## A

```C++
#include<iostream>
using namespace std;
const int N = 150;
int a[N];
int main(){
    int n;
    cin >> n;
    for(int i = 1;i<=n;i++){
        cin >> a[i];
    }
    int x;
    cin >> x ;
    cout << a[x] << endl;
    return 0;
}
```

## B

```C++
#include<iostream>
using namespace std;
const int N = 2e5+10;
int a[N][N];
int n;
int main(){
    cin >> n;
    int idx = 1;
    while(n-->0){
        int t;
        cin >> t;
        for(int i=1;i<=t;i++)
            cin >> a[idx][i];
        idx++;
    }
    int x,y;
    cin >> x >> y;
    cout << a[x][y] << endl;
    return 0;
}
```
